syn_fn_def(
  name: fn_symbol(branch),
  params: [
    (type: type_ref(type_symbol(boolexpr)), var: var(c)),
    (
      type: seq_type(nonempty: false, elem_type: type_ref(type_symbol(instr))),
      var: var(t)
    ),
    (
      type: seq_type(nonempty: false, elem_type: type_ref(type_symbol(instr))),
      var: var(f)
    )
  ],
  local_fns: [],
  expr: if_expr(
    else: tag_obj_expr(
      obj: map_expr(
        {
          (value: const_or_var(t), key: object(when_false)),
          (value: const_or_var(f), key: object(when_true)),
          (
            value: fn_call(
              name: fn_symbol(neg),
              params: [const_or_var(c)],
              named_params: []
            ),
            key: object(cond)
          )
        }
      ),
      tag: object(branch)
    ),
    branches: [
      (
        expr: tag_obj_expr(
          obj: map_expr(
            {
              (value: const_or_var(t), key: object(when_true)),
              (value: const_or_var(c), key: object(cond)),
              (value: const_or_var(f), key: object(when_false))
            }
          ),
          tag: object(branch)
        ),
        cond: or(
          left: neq(left: const_or_var(t), right: seq_expr(head: [])),
          right: eq(left: const_or_var(f), right: seq_expr(head: []))
        )
      )
    ]
  ),
  res_type: type_ref(type_symbol(instr))
)
