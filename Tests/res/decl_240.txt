syn_fn_def(
  name: fn_symbol(includes),
  params: [
    (type: type_ref(type_symbol(objpartset)), var: var(ps)),
    (type: type_ref(type_symbol(objpart)), var: var(p))
  ],
  local_fns: [],
  expr: do_expr(
    [
      if_stmt(
        else: [],
        branches: [
          (
            cond: fn_call(
              name: fn_symbol(in),
              params: [const_or_var(p), const_or_var(ps)],
              named_params: []
            ),
            body: [return_stmt(object(true))]
          )
        ]
      ),
      if_stmt(
        else: [],
        branches: [
          (
            cond: membership(
              type: union_type(
                {
                  tag_type(
                    obj_type: type_ref(type_symbol(atom)),
                    tag_type: symb_type(object(symbol))
                  )
                }
              ),
              obj: const_or_var(p)
            ),
            body: [
              return_stmt(
                fn_call(
                  name: fn_symbol(in),
                  params: [object(symbols), const_or_var(ps)],
                  named_params: []
                )
              )
            ]
          )
        ]
      ),
      if_stmt(
        else: [],
        branches: [
          (
            cond: membership(
              type: union_type(
                {
                  tag_type(
                    obj_type: type_ref(type_symbol(atom)),
                    tag_type: symb_type(object(tagged_obj))
                  )
                }
              ),
              obj: const_or_var(p)
            ),
            body: [
              return_stmt(
                fn_call(
                  name: fn_symbol(in),
                  params: [object(tagged_objs), const_or_var(ps)],
                  named_params: []
                )
              )
            ]
          )
        ]
      ),
      return_stmt(object(false))
    ]
  ),
  res_type: type_ref(type_symbol(bool))
)
