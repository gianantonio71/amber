  type: union_type(

      tag_type(
        obj_type: seq_type(nonempty: true, elem_type: type_ref(type_symbol(type))),
        tag_type: symb_type(object(fixed_seq_type))
      ),
      symb_type(object(empty_seq_type)),
      tag_type(
        obj_type: tuple_type(
          {
            (
              type: type_ref(type_symbol(bool)),
              optional: false,
              label: object(nonempty)
            ),
            (
              type: type_ref(type_symbol(type)),
              optional: false,
              label: object(elem_type)
            )
          }
        ),
        tag_type: symb_type(object(seq_type))
      )
    }
  ),
  name: type_symbol(seqtype)
)
